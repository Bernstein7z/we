identity_function

            
function identity_function(value) {
    return function () {
        return value;
    };
}

addf

            
function addf(x) {
    return function (y) {
        return x + y;
    };
}

applyf

            
function applyf(func) {
    return function (x) {
        return function (y) {
            return func(x, y);
        };
    };
}

curry

            
function curry(func, value1) {
    return function (value2) {
        return func(value1, value2)
    }
}

curry

            
const inc = curry(addf(1), 3);
inc();

methodize

            
function methodize(func) {
    return function (y) {
        return func(this, y);
    };
}

demethodize

            
function demethodize(method) {
    return function (x, y) {
        return method.call(x, y);
    };
}

twice

            
function twice(func) {
    return function (x) {
        return func(x, x);
    };
}

composeu

            
function composeu(innerFunc, outerFunc) {
    return function (value) {
        return outerFunc(innerFunc(value));
    };
}

composeb

            
function composeb(innerFunc, outerFunc) {
    return function (x, y, z) {
        return outerFunc(innerFunc(x, y), z);
    };
}

once

            
function once(func) {
    let count = 0;
    return function (...rest) {
        if (count === 1) return new Error("only run once.");
        count++;
        return func(rest);
    };
}
