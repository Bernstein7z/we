function applyf(func) {
    return function (x) {
        return function (y) {
            return func(x, y);
        };
    };
}

function fib(last) {
    var a = 0, b = 1, i = 0;

    while (i < last) {
        var temp = a;
        a = b;
        b += temp;
        i++;
    }

    return b;
}
